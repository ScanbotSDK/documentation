165 item.rawBytes = std::vector<unsigned char>(cRawBytes, cRawBytes + rawBytesLength);
166
167 if (scanbotsdk_barcode_result_get_format(barcode_result, barcode_idx,
168 &item.format) != SCANBOTSDK_OK) {
169 return {};

173 return {};
